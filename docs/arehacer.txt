-- 0. ELIMINAR USUARIOS PROBLEMÁTICOS (LIMPIEZA)
-- Cambia los correos abajo si necesitas eliminar cuentas específicas para re-crearlas.
DO $$
DECLARE
    emails_to_delete TEXT[] := ARRAY['ejemplo@correo.com', 'otro@correo.com']; -- AGREGA AQUÍ LOS CORREOS
    e TEXT;
BEGIN
    FOREACH e IN ARRAY emails_to_delete
    LOOP
        DELETE FROM auth.users WHERE email = e;
    END LOOP;
END $$;

-- 1. EXTENSIONES Y FUNCIONES DE APOYO
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- Función segura para verificar si es Admin (EVITA RECURSIVIDAD RLS)
-- El error "Database error querying schema" ocurre cuando una política se consulta a sí misma infinitamente.
CREATE OR REPLACE FUNCTION public.check_is_admin()
RETURNS BOOLEAN AS $$
BEGIN
    RETURN EXISTS (
        SELECT 1 FROM public.directory_users
        WHERE id = auth.uid() AND role = 'Admin'
    );
END;
$$ LANGUAGE plpgsql SECURITY DEFINER SET search_path = public;

-- 2. TABLAS DEL SISTEMA (IDEMPOTENTE)

-- Tabla de Usuarios del Directorio
CREATE TABLE IF NOT EXISTS public.directory_users (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    name TEXT UNIQUE NOT NULL,
    role TEXT DEFAULT 'Owner' CHECK (role IN ('Owner', 'Admin')),
    is_active BOOLEAN DEFAULT TRUE,
    max_businesses INTEGER DEFAULT 1,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()

    order_bg_color TEXT DEFAULT '#f2a04a',

-- 4. DISPARADORES (TRIGGERS) PARA AUTOMATIZACIÓN

-- Sincronizar auth.users con public.directory_users al registrarse
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO public.directory_users (id, name, role)
    VALUES (
        NEW.id,
        COALESCE(NEW.raw_user_meta_data->>'username', NEW.raw_user_meta_data->>'full_name', SPLIT_PART(NEW.email, '@', 1)),
        'Owner'
    )
    ON CONFLICT (id) DO UPDATE SET
        name = COALESCE(EXCLUDED.name, public.directory_users.name);
    RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER SET search_path = public;

DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
    AFTER INSERT ON auth.users
    FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();
