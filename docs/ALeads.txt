-- 1. ACTUALIZAR ROLES DISPONIBLES EN USUARIOS
-- Primero eliminamos la restricción antigua (si existe) y agregamos la nueva
ALTER TABLE public.usuarios DROP CONSTRAINT IF EXISTS usuarios_role_check;
ALTER TABLE public.usuarios ADD CONSTRAINT usuarios_role_check CHECK (role IN ('Owner', 'Admin', 'Ventas', 'CS'));

-- 2. TABLA DE LEADS (PROSPECTOS)
CREATE TABLE IF NOT EXISTS public.leads (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    salesperson_id UUID REFERENCES public.usuarios(id) ON DELETE CASCADE,
    business_name TEXT NOT NULL,
    owner_name TEXT,
    visit_date DATE DEFAULT CURRENT_DATE,
    phone TEXT,
    social_media JSONB DEFAULT '{}'::jsonb,
    address TEXT,
    status TEXT DEFAULT 'No Visitados' CHECK (status IN ('No Visitados', 'Visitados', 'Seguimiento', 'Venta', 'Agendados', 'No Interesados')),
    notes TEXT,
    image_url TEXT,
    scheduled_date TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 3. HABILITAR RLS
ALTER TABLE public.leads ENABLE ROW LEVEL SECURITY;

-- 4. FUNCION DE APOYO PARA ACCESO DE VENTAS/CS
CREATE OR REPLACE FUNCTION public.check_has_sales_access()
RETURNS BOOLEAN AS $$
BEGIN
    RETURN EXISTS (
        SELECT 1 FROM public.usuarios
        WHERE id = auth.uid() AND role IN ('Admin', 'Ventas', 'CS')
    );
END;
$$ LANGUAGE plpgsql SECURITY DEFINER SET search_path = public;

-- 5. POLÍTICAS DE RLS PARA LEADS
DROP POLICY IF EXISTS "Sales and Admin manage leads" ON public.leads;
CREATE POLICY "Sales and Admin manage leads" ON public.leads
    FOR ALL
    USING (public.check_has_sales_access());
