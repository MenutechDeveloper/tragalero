-- 1. EXTENSIONES
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- 2. FUNCIONES DE APOYO (EVITAR RECURSIVIDAD RLS)
CREATE OR REPLACE FUNCTION public.check_is_admin()
RETURNS BOOLEAN AS $$
BEGIN
    RETURN EXISTS (
        SELECT 1 FROM public.usuarios
        WHERE id = auth.uid() AND role = 'Admin'
    );
END;
$$ LANGUAGE plpgsql SECURITY DEFINER SET search_path = public;

-- 3. TABLA DE USUARIOS
CREATE TABLE IF NOT EXISTS public.usuarios (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    name TEXT NOT NULL UNIQUE,
    email TEXT UNIQUE,
    role TEXT DEFAULT 'Owner' CHECK (role IN ('Owner', 'Admin')),
    is_active BOOLEAN DEFAULT TRUE,
    max_businesses INTEGER DEFAULT 1,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 4. TABLA DE NEGOCIOS (RESTAURANTES)
CREATE TABLE IF NOT EXISTS public.businesses (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    owner_id UUID REFERENCES public.usuarios(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    category TEXT DEFAULT 'Restaurantes',
    city TEXT,
    state TEXT,
    zip TEXT,
    address_detail TEXT,
    latitude NUMERIC,
    longitude NUMERIC,
    logo_url TEXT,
    website TEXT,
    is_restaurant BOOLEAN DEFAULT TRUE,
    cuid TEXT,
    ruid TEXT,
    order_bg_color TEXT DEFAULT '#f2a04a',
    order_text_color TEXT DEFAULT '#ffffff',
    res_bg_color TEXT DEFAULT '#2f4854',
    res_text_color TEXT DEFAULT '#ffffff',
    has_reservation BOOLEAN DEFAULT FALSE,
    order_url TEXT,
    reservation_url TEXT,
    social_media JSONB DEFAULT '{}'::jsonb,
    hours JSONB DEFAULT '{}'::jsonb,
    is_visible BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 5. TABLA DE GALERIA
CREATE TABLE IF NOT EXISTS public.galeria (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id UUID REFERENCES public.usuarios(id) ON DELETE CASCADE,
    image_url TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 6. TABLA DE PROMOCIONES
CREATE TABLE IF NOT EXISTS public.promos (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id UUID REFERENCES public.usuarios(id) ON DELETE CASCADE,
    event_type TEXT NOT NULL,
    image_url TEXT,
    start_date DATE,
    end_date DATE,
    is_active BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    UNIQUE(user_id, event_type)
);

-- 7. TABLA DE CONFIGURE WEB (WEBSITES)
CREATE TABLE IF NOT EXISTS public.user_websites (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,

-- 8. TABLA DE TARJETAS DIGITALES

-- 9. TABLA BENTO GRID (PARA ADMIN CUSTOMIZATION - OPCIONAL AHORA)
CREATE TABLE IF NOT EXISTS public.bento_grid (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id UUID REFERENCES public.usuarios(id) ON DELETE CASCADE,
    label TEXT,
    link_url TEXT,
    col_span INTEGER DEFAULT 1,
    row_span INTEGER DEFAULT 1,
    bg_color TEXT,
    position INTEGER DEFAULT 0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()

-- 17. TRIGGER PARA SINCRONIZAR USUARIOS (AUTH -> PUBLIC)
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER AS $$
DECLARE
    base_name TEXT;
    final_name TEXT;
    counter INTEGER := 0;
BEGIN
    base_name := COALESCE(
        NEW.raw_user_meta_data->>'name',
        NEW.raw_user_meta_data->>'username',
        NEW.raw_user_meta_data->>'full_name',
        SPLIT_PART(NEW.email, '@', 1)
    );
    final_name := base_name;

    WHILE EXISTS (SELECT 1 FROM public.usuarios WHERE name = final_name AND id != NEW.id) LOOP
        counter := counter + 1;
        final_name := base_name || counter;
    END LOOP;

    INSERT INTO public.usuarios (id, name, email, role)
    VALUES (NEW.id, final_name, NEW.email, 'Owner')
    ON CONFLICT (id) DO UPDATE SET
        email = EXCLUDED.email,
        name = COALESCE(public.usuarios.name, EXCLUDED.name);

    RETURN NEW;
EXCEPTION WHEN OTHERS THEN
    RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER SET search_path = public;

DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();
