-- 1. ELIMINAR USUARIOS DE PRUEBA (Para que puedas registrarte de nuevo con el mismo correo)
-- ADVERTENCIA: Esto borrará los usuarios actuales en Auth y en la tabla usuarios.
-- Descomenta la línea de abajo y pon tu correo si quieres borrarlo específicamente:
-- DELETE FROM auth.users WHERE email = 'tu@email.com';

-- 2. LIMPIEZA TOTAL DE TRIGGERS
-- Este bloque busca y destruye CUALQUIER trigger que esté tocando la tabla de auth.
DO $$
DECLARE
    r RECORD;
BEGIN
    FOR r IN (SELECT trigger_name, event_object_table
              FROM information_schema.triggers
              WHERE event_object_schema = 'auth' AND event_object_table = 'users') LOOP
        EXECUTE format('DROP TRIGGER IF EXISTS %I ON auth.users', r.trigger_name);
    END LOOP;
END $$;

-- 3. TABLA USUARIOS (Garantizar estructura limpia)
CREATE TABLE IF NOT EXISTS public.usuarios (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    email TEXT,
    role TEXT DEFAULT 'Owner',
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 4. TRIGGER ULTRA-SIMPLIFICADO (Mínimo riesgo de Error 500)
-- Esta función solo inserta los datos básicos. Si falla, no detiene el registro.
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO public.usuarios (id, name, email, role)
  VALUES (
    NEW.id,
    COALESCE(NEW.raw_user_meta_data->>'name', NEW.raw_user_meta_data->>'username', 'Usuario'),
    NEW.email,
    'Owner'
  )
  ON CONFLICT (id) DO NOTHING;
  RETURN NEW;
EXCEPTION WHEN OTHERS THEN
  RETURN NEW; -- Ignorar errores para no romper el registro
END;
$$ LANGUAGE plpgsql SECURITY DEFINER SET search_path = public;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();

-- 5. SEGURIDAD RLS (RESET COMPLETO)
ALTER TABLE public.usuarios ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.businesses ENABLE ROW LEVEL SECURITY;

DO $$
DECLARE pol record;
BEGIN
    FOR pol IN (SELECT policyname, tablename FROM pg_policies WHERE schemaname = 'public') LOOP
        EXECUTE format('DROP POLICY IF EXISTS %I ON %I', pol.policyname, pol.tablename);
    END LOOP;
END $$;

-- Políticas ultra-permisivas para pruebas (luego las cerramos más)
CREATE POLICY "Acceso_Total_Usuarios" ON public.usuarios FOR ALL USING (TRUE);
CREATE POLICY "Acceso_Total_Negocios" ON public.businesses FOR ALL USING (TRUE);
