-- 1. LIMPIEZA DE TRIGGERS
DO $$
DECLARE
    r RECORD;
BEGIN
    FOR r IN (SELECT trigger_name
              FROM information_schema.triggers
              WHERE event_object_schema = 'auth' AND event_object_table = 'users') LOOP
        EXECUTE format('DROP TRIGGER IF EXISTS %I ON auth.users', r.trigger_name);
    END LOOP;
END $$;

-- 2. TABLA USUARIOS
CREATE TABLE IF NOT EXISTS public.usuarios (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    email TEXT,
    role TEXT DEFAULT 'Owner',
    is_active BOOLEAN DEFAULT TRUE,
    max_businesses INTEGER DEFAULT 1,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 3. TRIGGER DE SINCRONIZACIÓN (SECURITY DEFINER es clave aquí)
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO public.usuarios (id, name, email, role)
  VALUES (
    NEW.id,
    COALESCE(NEW.raw_user_meta_data->>'full_name', NEW.raw_user_meta_data->>'username', 'Usuario'),
    NEW.email,
    'Owner'
  )
  ON CONFLICT (id) DO UPDATE SET
    email = EXCLUDED.email;
  RETURN NEW;
EXCEPTION WHEN OTHERS THEN
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER SET search_path = public;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();

-- 4. POLÍTICAS RLS CORREGIDAS (Para evitar que te saque de la sesión)
ALTER TABLE public.usuarios ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.businesses ENABLE ROW LEVEL SECURITY;

DO $$
DECLARE pol record;
BEGIN
    FOR pol IN (SELECT policyname, tablename FROM pg_policies WHERE schemaname = 'public') LOOP
        EXECUTE format('DROP POLICY IF EXISTS %I ON %I', pol.policyname, pol.tablename);
    END LOOP;
END $$;

-- Política vital: Permitir que el usuario lea su propio perfil (Sin esto, getLoggedInUser falla)
CREATE POLICY "Permitir_lectura_propia"
ON public.usuarios FOR SELECT
USING (auth.uid() = id);

-- Política para que los admins vean todo
CREATE POLICY "Admins_ver_todo"
ON public.usuarios FOR SELECT
USING (
  EXISTS (
    SELECT 1 FROM public.usuarios
    WHERE id = auth.uid() AND role = 'Admin'
  )
);

-- Negocios (Permitir lectura pública y gestión propia)
CREATE POLICY "Negocios_Ver_Publico" ON public.businesses FOR SELECT USING (TRUE);
CREATE POLICY "Negocios_Gestion_Dueno" ON public.businesses FOR ALL USING (auth.uid() = owner_id);
