-- 1. EXTENSIONES Y FUNCIONES DE APOYO
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- Función segura para verificar si es Admin (EVITA RECURSIVIDAD RLS)
CREATE OR REPLACE FUNCTION public.check_is_admin()
RETURNS BOOLEAN AS $$
BEGIN
    RETURN EXISTS (
        SELECT 1 FROM public.usuarios
        WHERE id = auth.uid() AND role = 'Admin'
    );
END;
$$ LANGUAGE plpgsql SECURITY DEFINER SET search_path = public;

-- 2. TABLA DE USUARIOS
CREATE TABLE IF NOT EXISTS public.usuarios (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    email TEXT UNIQUE,
    role TEXT DEFAULT 'Owner' CHECK (role IN ('Owner', 'Admin')),
    is_active BOOLEAN DEFAULT TRUE,
    max_businesses INTEGER DEFAULT 1,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Asegurar restricción de nombre único
DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_constraint WHERE conname = 'usuarios_name_key') THEN
        ALTER TABLE public.usuarios ADD CONSTRAINT usuarios_name_key UNIQUE (name);
    END IF;
END $$;

-- 3. HABILITAR RLS
ALTER TABLE public.usuarios ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.businesses ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.bento_grid ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.digital_cards ENABLE ROW LEVEL SECURITY;

-- 4. DISPARADOR DE REGISTRO (TRIGGER) - VERSION ULTRA ROBUSTA
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER AS $$
DECLARE
    base_name TEXT;
    final_name TEXT;
    counter INTEGER := 0;
BEGIN
    -- 1. Extraer nombre de metadatos o email
    base_name := COALESCE(
        NEW.raw_user_meta_data->>'name',
        NEW.raw_user_meta_data->>'username',
        NEW.raw_user_meta_data->>'full_name',
        SPLIT_PART(NEW.email, '@', 1)
    );
    final_name := base_name;

    -- 2. Asegurar unicidad del nombre
    WHILE EXISTS (SELECT 1 FROM public.usuarios WHERE name = final_name AND id != NEW.id) LOOP
        counter := counter + 1;
        final_name := base_name || counter;
    END LOOP;

    -- 3. Insertar con manejo de conflictos
    INSERT INTO public.usuarios (id, name, email, role)
    VALUES (NEW.id, final_name, NEW.email, 'Owner')
    ON CONFLICT (id) DO UPDATE SET
        email = EXCLUDED.email,
        name = COALESCE(public.usuarios.name, EXCLUDED.name);

    RETURN NEW;
EXCEPTION WHEN OTHERS THEN
    -- Si algo falla aquí, al menos logueamos o dejamos que continúe
    -- No bloqueamos el registro de Auth si el perfil falla (opcional según necesidad)
    RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER SET search_path = public;

DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();

-- 5. POLÍTICAS DE RLS
DROP POLICY IF EXISTS "Public profile view" ON public.usuarios;
CREATE POLICY "Public profile view" ON public.usuarios FOR SELECT USING (TRUE);

DROP POLICY IF EXISTS "Self manage profile" ON public.usuarios;
CREATE POLICY "Self manage profile" ON public.usuarios FOR ALL USING (auth.uid() = id);

DROP POLICY IF EXISTS "Admin full access users" ON public.usuarios;
CREATE POLICY "Admin full access users" ON public.usuarios FOR ALL USING (public.check_is_admin());
