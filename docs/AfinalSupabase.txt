-- 1. ACTUALIZAR TABLA DE USUARIOS PARA SOPORTAR EMAIL
ALTER TABLE public.usuarios ADD COLUMN IF NOT EXISTS email TEXT UNIQUE;

-- 2. HABILITAR RLS EN TODAS LAS TABLAS
ALTER TABLE public.usuarios ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.businesses ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.bento_grid ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.digital_cards ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.user_websites ENABLE ROW LEVEL SECURITY;

-- 3. FUNCIÓN Y TRIGGER PARA SINCRONIZACIÓN AUTOMÁTICA
-- Esto crea un perfil en la tabla 'public.usuarios' cada vez que alguien se registra en Auth.
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO public.usuarios (id, name, email, role)
  VALUES (
    NEW.id,
    COALESCE(NEW.raw_user_meta_data->>'name', NEW.raw_user_meta_data->>'username', 'Usuario ' || NEW.id),
    NEW.email,
    'Owner'
  )
  ON CONFLICT (id) DO UPDATE SET
    email = EXCLUDED.email,
    name = COALESCE(EXCLUDED.name, public.usuarios.name);
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Borramos el trigger si ya existe para evitar errores al re-ejecutar
DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();

-- 4. POLÍTICAS DE RLS PARA 'usuarios'
DROP POLICY IF EXISTS "Los usuarios pueden ver su propio perfil" ON public.usuarios;
CREATE POLICY "Los usuarios pueden ver su propio perfil"
ON public.usuarios FOR SELECT
USING (auth.uid() = id);

DROP POLICY IF EXISTS "Los administradores pueden ver todos los perfiles" ON public.usuarios;
CREATE POLICY "Los administradores pueden ver todos los perfiles"
ON public.usuarios FOR SELECT
USING (
  EXISTS (
    SELECT 1 FROM public.usuarios
    WHERE id = auth.uid() AND role = 'Admin'
  )
);

DROP POLICY IF EXISTS "Los administradores pueden actualizar perfiles" ON public.usuarios;
CREATE POLICY "Los administradores pueden actualizar perfiles"
ON public.usuarios FOR UPDATE
USING (
  EXISTS (
    SELECT 1 FROM public.usuarios
    WHERE id = auth.uid() AND role = 'Admin'
  )
);
